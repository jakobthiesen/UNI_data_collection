\section{Scope}
This guide applies to all students and researchers collecting experimental datasets in the APMS laboratories, covering analog, digital, RF, and AI-related measurements.

\section{References}
\begin{itemize}[noitemsep]
    \item IEEE Data Description Principles
    \item ISO/IEC 17025: General requirements for the competence of testing and calibration laboratories
    \item BIPM GUM: \textit{Guide to the Expression of Uncertainty in Measurement}
    \item Instrument datasheets and calibration certificates
\end{itemize}

\section{Roles and Responsibilities}
\begin{itemize}[noitemsep]
    \item \textbf{Student:} Performs measurements according to procedure, logs data, and observes anomalies.
    \item \textbf{Supervisor:} Reviews experimental protocols, validates measurements, and approves datasets.
    \item \textbf{Lab Manager/Technical Mentor:} Ensures instruments are calibrated, procedures are up-to-date, and training is provided.
\end{itemize}

\section{Preparation Before Measurement}
\begin{enumerate}[label=\arabic*.]
    \item Define measurement goals and required accuracy.
    \item Select appropriate instruments and sensors, ensuring calibration and traceability.
    \item Check instrument specifications: range, linearity, noise, resolution, and sample rate.
    \item Draft experimental protocol:
    \begin{itemize}
        \item Step-by-step measurement plan
        \item Identify critical variables and sources of error
        \item Plan repetitions to quantify repeatability
    \end{itemize}
    \item Log calibration procedure and reference standards used.
\end{enumerate}

\section{During Measurement}
\begin{enumerate}[label=\arabic*.]
    \item Record date, time, experiment ID, and operator name.
    \item Log instrument settings, configurations, and environmental conditions.
    \item Observe and record anomalies, drift, or unexpected behavior.
    \item Repeat measurements to quantify variability and reduce bias.
\end{enumerate}

\section{After Measurement}
\begin{enumerate}[label=\arabic*.]
    \item Preserve raw data intact in designated open formats (e.g., HDF5, CSV).
    \item Perform initial data cleaning if necessary (document any modifications).
    \item Estimate measurement uncertainty:
    \begin{itemize}
        \item Identify sources (instrument, method, environment, operator)
        \item Use Type A (statistical) and Type B (systematic) methods
        \item Combine uncertainties according to GUM guidelines
    \end{itemize}
    \item Validate data completeness and quality.
    \item Document assumptions, limitations, and known biases.
\end{enumerate}

\section{Measurement Procedure Template}
\subsection*{Title and Purpose}
\textit{E.g., Calibration and Data Collection Procedure for RF Power Measurements}

\subsection*{Equipment List}
Include serial numbers, calibration status, and traceability references.

\subsection*{Environmental Conditions}
Temperature, humidity, vibration, and EM interference considerations.

\subsection*{Step-by-Step Procedure}
\begin{enumerate}[label=\arabic*.]
    \item Pre-measurement instrument warm-up and verification.
    \item Connection diagram setup.
    \item Measurement steps with instrument settings.
    \item Repetition and randomization of measurements.
    \item Observation logging (drift, noise, anomalies).
\end{enumerate}

\subsection*{Uncertainty Evaluation}
Provide formulas or methods for combining uncertainties from all sources.

\subsection*{Data Logging and Documentation}
Templates for experiment ID, date/time, instrument settings, raw data path, and observation notes.

\subsection*{Verification and Validation}
Baseline checks, cross-checks with known standards, and repeatability verification.

\section{Key Principles}
\begin{itemize}[noitemsep]
    \item Traceability is essential: link measurements to references.
    \item Quantify uncertainty rather than assuming it.
    \item Document rigorously for reproducibility.
    \item Apply critical thinking: check that results make sense physically.
\end{itemize}
